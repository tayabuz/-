105  public static async Task<T> ReadObjectFromXmlFileAsync<T>(string filename)
106  {
107  T objectFromXml = default(T);
108  var serializer = new XmlSerializer(typeof(T));
109  StorageFolder folder = ApplicationData.Current.LocalFolder;
110  StorageFile file = await folder.GetFileAsync(filename);
111  Stream stream = await file.OpenStreamForReadAsync();
112  objectFromXml = (T)serializer.Deserialize(stream);
113  stream.Dispose();
114  return objectFromXml;
115  }
